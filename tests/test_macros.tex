    \centering\footnotesize
    \caption{#2}
    \label{#3}
    \renewcommand{\arraystretch}{1.15}
    \csvreader[
      tabular={llll},
      table head={\toprule Category & Component & Part~\# & Notes / key specs \\\midrule},
      table foot=\bottomrule,
      late after line=\\,
    ]{#1}{1=\colone,2=\coltwo,3=\colthree,4=\colfour}{%
      \colone & \coltwo & \colthree & \colfour
    }
  \end{table*}
}
